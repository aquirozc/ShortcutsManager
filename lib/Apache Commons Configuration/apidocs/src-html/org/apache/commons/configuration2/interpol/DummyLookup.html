017package org.apache.commons.configuration2.interpol;
018
019/**
020 * <p>
021 * A simple dummy {@code Lookup} implementation.
022 * </p>
023 * <p>
024 * This implementation always returns <b>null</b> for a passed in variable indicating that it cannot resolve that
025 * variable. This is useful for instance as an application of the <em>null object</em> pattern.
026 * </p>
027 * <p>
028 * This class does not define any state, therefore a single instance can be shared. To enforce usage of only a single
029 * instance this class is actually an enumeration.
030 * </p>
031 *
032 * @since 2.0
033 */
034public enum DummyLookup implements Lookup {
035    /** The single instance of this class. */
036    INSTANCE;
037
038    /**
039     * {@inheritDoc} This implementation always returns <b>null</b>.
040     */
041    @Override
042    public Object lookup(final String variable) {
043        return null;
044    }
045}
