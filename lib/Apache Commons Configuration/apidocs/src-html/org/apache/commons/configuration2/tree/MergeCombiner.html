017package org.apache.commons.configuration2.tree;
018
019import java.util.ArrayList;
020import java.util.HashMap;
021import java.util.LinkedList;
022import java.util.List;
023import java.util.Map;
024import java.util.Objects;
025
026/**
027 * <p>
028 * A specialized implementation of the {@code NodeCombiner} interface that performs a merge from two passed in node
029 * hierarchies.
030 * </p>
031 * <p>
032 * This combiner performs the merge using a few rules:
033 * </p>
034 * <ol>
035 * <li>Nodes can be merged when attributes that appear in both have the same value.</li>
036 * <li>Only a single node in the second file is considered a match to the node in the first file.</li>
037 * <li>Attributes in nodes that match are merged.
038 * <li>Nodes in both files that do not match are added to the result.</li>
039 * </ol>
040 *
041 * @since 1.7
042 */
043public class MergeCombiner extends NodeCombiner {
044    /**
045     * Combines the given nodes to a new union node.
046     *
047     * @param node1 the first source node
048     * @param node2 the second source node
049     * @return the union node
050     */
051
052    @Override
053    public ImmutableNode combine(final ImmutableNode node1, final ImmutableNode node2) {
054        final ImmutableNode.Builder result = new ImmutableNode.Builder();
055        result.name(node1.getNodeName());
056        result.value(node1.getValue());
057        addAttributes(result, node1, node2);
058
059        // Check if nodes can be combined
060        final List<ImmutableNode> children2 = new LinkedList<>(node2.getChildren());
061        for (final ImmutableNode child1 : node1) {
062            final ImmutableNode child2 = canCombine(node2, child1, children2);
063            if (child2 != null) {
064                result.addChild(combine(child1, child2));
065                children2.remove(child2);
066            } else {
067                result.addChild(child1);
068            }
069        }
070
071        // Add remaining children of node 2
072        for (final ImmutableNode c : children2) {
073            result.addChild(c);
074        }
075        return result.create();
076    }
077
078    /**
079     * Handles the attributes during a combination process. First all attributes of the first node will be added to the
080     * result. Then all attributes of the second node, which are not contained in the first node, will also be added.
081     *
082     * @param result the builder for the resulting node
083     * @param node1 the first node
084     * @param node2 the second node
085     */
086    protected void addAttributes(final ImmutableNode.Builder result, final ImmutableNode node1, final ImmutableNode node2) {
087        final Map<String, Object> attributes = new HashMap<>(node1.getAttributes());
088        for (final Map.Entry<String, Object> e : node2.getAttributes().entrySet()) {
089            if (!attributes.containsKey(e.getKey())) {
090                attributes.put(e.getKey(), e.getValue());
091            }
092        }
093        result.addAttributes(attributes);
094    }
095
096    /**
097     * Tests if the first node can be combined with the second node. A node can only be combined if its attributes are all
098     * present in the second node and they all have the same value.
099     *
100     * @param node2 the second node
101     * @param child the child node (of the first node)
102     * @param children2 the children of the 2nd node
103     * @return a child of the second node, with which a combination is possible
104     */
105    protected ImmutableNode canCombine(final ImmutableNode node2, final ImmutableNode child, final List<ImmutableNode> children2) {
106        final Map<String, Object> attrs1 = child.getAttributes();
107        final List<ImmutableNode> nodes = new ArrayList<>();
108
109        final List<ImmutableNode> children = HANDLER.getChildren(node2, child.getNodeName());
110        for (final ImmutableNode node : children) {
111            if (matchAttributes(attrs1, node)) {
112                nodes.add(node);
113            }
114        }
115
116        if (nodes.size() == 1) {
117            return nodes.get(0);
118        }
119        if (nodes.size() > 1 && !isListNode(child)) {
120            for (final ImmutableNode node : nodes) {
121                children2.remove(node);
122            }
123        }
124
125        return null;
126    }
127
128    /**
129     * Checks whether the attributes of the passed in node are compatible.
130     *
131     * @param attrs1 the attributes of the first node
132     * @param node the 2nd node
133     * @return a flag whether these nodes can be combined regarding their attributes
134     */
135    private static boolean matchAttributes(final Map<String, Object> attrs1, final ImmutableNode node) {
136        final Map<String, Object> attrs2 = node.getAttributes();
137        for (final Map.Entry<String, Object> e : attrs1.entrySet()) {
138            if (attrs2.containsKey(e.getKey()) && !Objects.equals(e.getValue(), attrs2.get(e.getKey()))) {
139                return false;
140            }
141        }
142        return true;
143    }
144}
