017package org.apache.commons.configuration2.resolver;
018
019import java.io.IOException;
020import java.io.InputStream;
021import java.net.URL;
022import java.net.URLConnection;
023import java.util.HashMap;
024import java.util.Map;
025
026import org.xml.sax.EntityResolver;
027import org.xml.sax.InputSource;
028import org.xml.sax.SAXException;
029
030/**
031 * The DefaultEntityResolver used by XML Configurations.
032 *
033 * @since 1.7
034 */
035public class DefaultEntityResolver implements EntityResolver, EntityRegistry {
036    /** Stores a map with the registered public IDs. */
037    private final Map<String, URL> registeredEntities = new HashMap<>();
038
039    /**
040     * <p>
041     * Registers the specified URL for the specified public identifier.
042     * </p>
043     * <p>
044     * This implementation maps {@code PUBLICID}'s to URLs (from which the resource will be loaded). A common use case for
045     * this method is to register local URLs (possibly computed at runtime by a class loader) for DTDs and Schemas. This
046     * allows the performance advantage of using a local version without having to ensure every {@code SYSTEM} URI on every
047     * processed XML document is local. This implementation provides only basic functionality. If more sophisticated
048     * features are required, either calling {@code XMLConfiguration.setDocumentBuilder(DocumentBuilder)} to set a custom
049     * {@code DocumentBuilder} (which also can be initialized with a custom {@code EntityResolver}) or creating a custom
050     * entity resolver and registering it with the XMLConfiguration is recommended.
051     * </p>
052     *
053     * @param publicId Public identifier of the Entity to be resolved
054     * @param entityURL The URL to use for reading this Entity
055     * @throws IllegalArgumentException if the public ID is undefined
056     */
057    @Override
058    public void registerEntityId(final String publicId, final URL entityURL) {
059        if (publicId == null) {
060            throw new IllegalArgumentException("Public ID must not be null!");
061        }
062        getRegisteredEntities().put(publicId, entityURL);
063    }
064
065    /**
066     * Resolves the requested external entity. This is the default implementation of the {@code EntityResolver} interface.
067     * It checks the passed in public ID against the registered entity IDs and uses a local URL if possible.
068     *
069     * @param publicId the public identifier of the entity being referenced
070     * @param systemId the system identifier of the entity being referenced
071     * @return an input source for the specified entity
072     * @throws org.xml.sax.SAXException if a parsing exception occurs
073     */
074    @SuppressWarnings("resource") // The stream is managed by the InputSource returned by this method.
075    @Override
076    public InputSource resolveEntity(final String publicId, final String systemId) throws SAXException {
077        // Has this system identifier been registered?
078        URL entityURL = null;
079        if (publicId != null) {
080            entityURL = getRegisteredEntities().get(publicId);
081        }
082
083        if (entityURL != null) {
084            // Obtain an InputSource for this URL. This code is based on the
085            // createInputSourceFromURL() method of Commons Digester.
086            try {
087                final URLConnection connection = entityURL.openConnection();
088                connection.setUseCaches(false);
089                final InputStream stream = connection.getInputStream();
090                final InputSource source = new InputSource(stream);
091                source.setSystemId(entityURL.toExternalForm());
092                return source;
093            } catch (final IOException e) {
094                throw new SAXException(e);
095            }
096        }
097        // default processing behavior
098        return null;
099    }
100
101    /**
102     * Returns a map with the entity IDs that have been registered using the {@code registerEntityId()} method.
103     *
104     * @return a map with the registered entity IDs
105     */
106    @Override
107    public Map<String, URL> getRegisteredEntities() {
108        return registeredEntities;
109    }
110}
